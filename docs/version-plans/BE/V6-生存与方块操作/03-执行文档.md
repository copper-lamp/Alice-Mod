### 3.1 `eat` 实现流程

```typescript
// SurvivalEngine.eat()
async eat(foodName?: string): Promise<EatResult> {
  const inventory = this.inventoryEngine.list();
  let targetFood: ItemSlot | null = null;

  if (foodName) {
    targetFood = this.inventoryEngine.find(foodName);
    if (!targetFood) return { success: false, error: 'NO_FOOD' };
    if (!isFood(targetFood.name)) return { success: false, error: 'CANNOT_EAT' };
  } else {
    targetFood = this.foodSelector.selectBest(inventory);
    if (!targetFood) return { success: false, error: 'NO_FOOD' };
  }

  this.inventoryEngine.selectSlot(targetFood.slot);
  const startHunger = this.player.getHunger();
  const startSaturation = this.player.getSaturation();

  // LLSE 模拟食用：模拟右键使用
  this.player.simulateUseItem();
  await waitForEatComplete(this.player, 10000);

  BotManager.saveInventory(this.botName);

  return {
    success: true,
    item: targetFood.name,
    hunger_restored: this.player.getHunger() - startHunger,
    saturation_restored: this.player.getSaturation() - startSaturation,
  };
}
```

### 3.2 `sleep` 实现流程

```typescript
// SurvivalEngine.sleep()
async sleep(action: 'sleep' | 'wake', bedPos?: Vec3, maxWaitMs = 60000): Promise<SleepResult> {
  if (action === 'wake') {
    this.player.wake();
    return { success: true, slept_duration: 0 };
  }

  const bed = bedPos
    ? this.world.getBlock(bedPos.x, bedPos.y, bedPos.z)
    : this.bedFinder.findNearest(this.player.getPosition(), 32);

  if (!bed || !isBed(bed.name)) return { success: false, error: 'NO_BED' };

  const moveResult = await aiEngine.moveTo(this.botName, bedPosToVec3(bed));
  if (!moveResult.success) return { success: false, error: moveResult.reason };

  const check = this.bedFinder.checkSleepConditions(bedPos);
  if (!check.ok) return { success: false, error: check.reason };

  const sleepStart = Date.now();
  this.player.sleep(bedPos);

  await waitUntil(
    () => !this.player.isSleeping() || isDayTime(this.world),
    maxWaitMs,
    500
  );

  if (this.player.isSleeping()) this.player.wake();

  return {
    success: true,
    slept_duration: Date.now() - sleepStart,
    time_when_wake: this.world.getTime(),
  };
}
```

### 3.3 `mine_block` 实现流程

```typescript
// BlockOperationEngine.mineBlock()
async mineBlock(pos: Vec3): Promise<MineResult> {
  // 1. 移动靠近
  const moveResult = await aiEngine.moveTo(this.botName, pos, { distance: 5 });
  if (!moveResult.success) return { success: false, error: moveResult.reason };

  // 2. 获取方块
  const block = this.world.getBlock(pos.x, pos.y, pos.z);
  if (!block || block.name === 'air') {
    return { success: true, block: 'air', drops: [], duration_ms: 0 };
  }
  if (isUnbreakable(block.name)) {
    return { success: false, error: 'BLOCK_UNBREAKABLE' };
  }

  // 3. 选择工具
  const toolRec = this.toolSelector.selectToolForBlock(block.name, this.inventoryEngine.list());
  if (!toolRec.canHandMine && !toolRec.toolSlot) {
    return { success: false, error: 'NO_SUITABLE_TOOL' };
  }
  if (toolRec.toolSlot !== null) {
    this.inventoryEngine.selectSlot(toolRec.toolSlot);
  }

  // 4. 耐久检查
  const toolItem = this.inventoryEngine.getSelectedItem();
  if (toolItem && toolItem.damage + expectedDamage(block) >= toolItem.maxDamage) {
    const alt = this.toolSelector.findAlternativeTool(block.name, this.inventoryEngine.list());
    if (alt) this.inventoryEngine.selectSlot(alt.slot);
    else return { success: false, error: 'TOOL_DURABILITY_LOW' };
  }

  // 5. 执行挖掘
  const start = Date.now();
  this.player.simulateDestroyBlock(pos);
  await waitForBlockBreak(this.world, pos, 15000);

  // 6. 校验与保存
  const broken = this.blockValidator.confirmBroken(pos);
  BotManager.saveInventory(this.botName);

  return {
    success: broken,
    block: block.name,
    drops: [], // 掉落物由 BDS 自然生成，假人自动拾取
    tool_damage: expectedDamage(block),
    duration_ms: Date.now() - start,
    error: broken ? undefined : 'BLOCK_BREAK_TIMEOUT',
  };
}
```

### 3.4 `place_block` 实现流程

```typescript
// BlockOperationEngine.placeBlock()
async placeBlock(pos: Vec3, blockName: string): Promise<PlaceResult> {
  // 1. 材料检查与替代
  let material = blockName;
  let slot = this.inventoryEngine.find(blockName);
  if (!slot && config.allowAlternativeMaterials) {
    const alt = this.toolSelector.selectAlternativeBlock(blockName, this.inventoryEngine.list());
    if (alt) { material = alt; slot = this.inventoryEngine.find(alt); }
  }
  if (!slot) return { success: false, error: 'ITEM_NOT_FOUND' };

  // 2. 移动靠近
  const moveResult = await aiEngine.moveTo(this.botName, pos, { distance: 3 });
  if (!moveResult.success) return { success: false, error: moveResult.reason };

  // 3. 邻接面选择
  const face = this.blockValidator.findPlacementFace(pos);
  if (!face) return { success: false, error: 'PLACE_BLOCKED' };

  // 4. 放置
  this.inventoryEngine.selectSlot(slot.slot);
  this.player.simulatePlaceBlock(pos, face);

  // 5. 验证
  await sleep(100);
  const placed = this.blockValidator.confirmPlaced(pos, material);
  BotManager.saveInventory(this.botName);

  return {
    success: placed,
    block: material,
    position: pos,
    error: placed ? undefined : 'PLACE_VERIFY_FAILED',
  };
}
```

### 3.5 `area_operation` 实现流程

```typescript
// BlockOperationEngine.areaOperation()
async areaOperation(
  mode: AreaMode,
  from: Vec3,
  to?: Vec3,
  blockName?: string,
  radius = 16
): Promise<AreaResult> {
  const queue = this.areaPlanner.buildQueue(mode, from, to, blockName, radius);

  if (queue.length > config.max_area_operation_blocks) {
    return { success: false, error: 'AREA_TOO_LARGE', total_blocks: queue.length };
  }

  let successCount = 0;
  let failCount = 0;
  const drops: Record<string, number> = {};
  const start = Date.now();

  for (let i = 0; i < queue.length; i++) {
    const pos = queue[i];
    let result: MineResult | PlaceResult;

    if (mode === 'fill') {
      result = await this.placeBlock(pos, blockName!);
    } else {
      result = await this.mineBlock(pos);
      if (mode === 'break' && result.success) {
        // 汇总掉落物：通过操作前后背包变化估算
        mergeDrops(drops, estimateDrops(result.block));
      }
    }

    if (result.success) successCount++; else failCount++;

    // 每 8 个操作 yield 一次
    if ((i + 1) % 8 === 0) await sleep(0);

    // 超时检查
    if (Date.now() - start > 120000) {
      return { success: false, error: 'OPERATION_TIMEOUT', total_blocks: queue.length, success_count: successCount, fail_count: failCount };
    }
  }

  return {
    success: failCount === 0,
    mode,
    total_blocks: queue.length,
    success_count: successCount,
    fail_count: failCount,
    duration_ms: Date.now() - start,
  };
}
```

## 第7章 风险与应对

| 风险 | 概率 | 影响 | 应对措施 |
|:----:|:----:|:----:|----------|
| LLSE `SimulatedPlayer` 食用/睡觉 API 不完整 | 中 | 高 |  fallback 到 `/effect` 或游戏内命令；记录实际 API 行为 |
| `simulatePlaceBlock` 对特殊方块（门、楼梯）方向控制不精确 | 中 | 中 | 对常见方块预定义放置方向；方向不敏感时接受默认 |
| `area_operation` 逐格执行导致大区域耗时过长 | 中 | 中 | 严格上限 256；超时返回部分结果；LLM 可拆分多次调用 |
| vein 模式 BFS 误入不可破坏方块边界 | 低 | 中 | BFS 遇到不可破坏方块自动停止该分支 |
| 食物营养值表与实际游戏版本不一致 | 低 | 低 | 使用游戏内测试校准；优先保证相对排序正确 |
| 多假人并发执行方块操作导致世界状态冲突 | 低 | 高 | 同一区域操作时增加轻量锁或 LLM 层调度避免 |
