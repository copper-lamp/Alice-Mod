### 3.1 `InGameToolTester` 入口示例

```typescript
// src/test/InGameToolTester.ts

import { ToolManager } from '../registry/tool-manager.js';
import { ToolContextImpl } from '../registry/tool-context.js';
import { BotManager } from '../bot/BotManager.js';
import { ToolMenuGui } from './ToolMenuGui.js';
import { ToolParamFormBuilder } from './ToolParamFormBuilder.js';
import { ToolResultRenderer } from './ToolResultRenderer.js';
import { ToolTestReport } from './ToolTestReport.js';
import type { PlayerSession } from './types.js';

export class InGameToolTester {
  private sessions = new Map<string, PlayerSession>();
  private report = new ToolTestReport();

  constructor(
    private toolManager: ToolManager,
    private config: GuiTestConfig,
  ) {}

  onCommand(player: Player): void {
    if (!this.checkPermission(player)) {
      player.sendText('§c需要 OP 权限或白名单才能使用测试工具');
      return;
    }

    const bots = BotManager.getAll().filter((b) => b.isOnline());
    if (bots.length === 0) {
      player.sendText('§c没有在线假人，请先创建并上线假人');
      return;
    }

    this.ensureSession(player, bots[0].name);
    ToolMenuGui.showMainMenu(player, (action) => this.handleMenuAction(player, action));
  }

  private checkPermission(player: Player): boolean {
    if (!this.config.enabled) return false;
    if (this.config.requireOp && !player.isOP()) return false;
    if (this.config.allowedPlayers.length > 0 &&
        !this.config.allowedPlayers.includes(player.xuid)) return false;
    return true;
  }

  private ensureSession(player: Player, defaultBot: string): void {
    if (!this.sessions.has(player.xuid)) {
      this.sessions.set(player.xuid, {
        activeBot: this.config.defaultTargetBot || defaultBot,
        pendingExecution: null,
      });
    }
  }

  private getSession(player: Player): PlayerSession {
    return this.sessions.get(player.xuid)!;
  }
}
```

### 3.3 `ToolParamFormBuilder` 解析与默认值

```typescript
// src/test/ToolParamFormBuilder.ts

import type { ToolMetadata } from '../registry/tool-module.types.js';
import { BetterCustomForm, BetterCustomFormResult } from '../../lib/LLSE-FakePlayer/Gui/BetterForms.js';

export class ToolParamFormBuilder {
  static build(player: Player, metadata: ToolMetadata): BetterCustomForm {
    const form = new BetterCustomForm(`§l测试: ${metadata.name}`);
    const schema = metadata.input_schema;
    const required = new Set(schema.required || []);

    // 危险操作提示
    if (['mine_block', 'place_block', 'area_operation'].includes(metadata.name)) {
      form.addLabel('warning', '§c注意：此工具会修改世界方块，请在测试区域使用。');
    }

    for (const [key, prop] of Object.entries(schema.properties || {})) {
      const title = `${key}${required.has(key) ? ' *' : ''}`;

      if (prop.enum) {
        const items = prop.enum.map((v) => String(v));
        const def = prop.default !== undefined ? items.indexOf(String(prop.default)) : 0;
        form.addDropdown(key, title, items, Math.max(0, def));
      } else if (prop.type === 'boolean') {
        form.addSwitch(key, title, prop.default ?? false);
      } else if (prop.type === 'number') {
        const def = this.getCoordinateDefault(player, key, prop.default);
        form.addInput(key, title, '', String(def));
      } else if (prop.type === 'string') {
        const def = this.getStringDefault(player, key, prop.default);
        form.addInput(key, title, '', def);
      } else if (prop.type === 'array') {
        form.addInput(key, title, '逗号分隔', '');
      } else if (prop.type === 'object' && prop.properties?.x) {
        form.addInput(`${key}.x`, `${title}.x`, '', String(this.getCoordinateDefault(player, 'x')));
        form.addInput(`${key}.y`, `${title}.y`, '', String(this.getCoordinateDefault(player, 'y')));
        form.addInput(`${key}.z`, `${title}.z`, '', String(this.getCoordinateDefault(player, 'z')));
      }
    }

    return form;
  }

  static parse(result: BetterCustomFormResult, metadata: ToolMetadata): Record<string, unknown> {
    const params: Record<string, unknown> = {};
    const schema = metadata.input_schema;

    for (const [key, prop] of Object.entries(schema.properties || {})) {
      if (prop.type === 'object' && prop.properties?.x) {
        params[key] = {
          x: Number(result.get(`${key}.x`)),
          y: Number(result.get(`${key}.y`)),
          z: Number(result.get(`${key}.z`)),
        };
      } else if (prop.type === 'number') {
        params[key] = Number(result.get(key));
      } else if (prop.type === 'boolean') {
        params[key] = Boolean(result.get(key));
      } else if (prop.type === 'array') {
        const raw = String(result.get(key));
        params[key] = raw ? raw.split(',').map((s) => s.trim()) : [];
      } else {
        params[key] = result.get(key);
      }
    }

    return params;
  }

  private static getCoordinateDefault(player: Player, key: string, fallback?: number): number {
    if (key === 'x') return Math.floor(player.pos.x);
    if (key === 'y') return Math.floor(player.pos.y);
    if (key === 'z') return Math.floor(player.pos.z);
    return fallback ?? 0;
  }

  private static getStringDefault(player: Player, key: string, fallback?: string): string {
    if (key === 'block_name') return 'stone';
    if (key === 'item_name') {
      const hand = player.getHand();
      return hand.isNull() ? 'apple' : hand.name;
    }
    return String(fallback ?? '');
  }
}
```

### 3.4 工具执行与结果展示

```typescript
// src/test/InGameToolTester.ts（续）

async executeTool(player: Player, toolName: string, params: Record<string, unknown>): Promise<void> {
  const session = this.getSession(player);
  if (session.pendingExecution) {
    player.sendText('§c已有工具正在执行，请等待完成');
    return;
  }

  const bot = BotManager.get(session.activeBot);
  if (!bot || !bot.isOnline()) {
    ToolResultRenderer.renderError(player, `目标假人 ${session.activeBot} 不在线`);
    return;
  }

  player.sendText(`§e[测试] 正在执行 ${toolName} ...`);

  session.pendingExecution = this.doExecute(player, toolName, params);
  await session.pendingExecution;
  session.pendingExecution = null;
}

private async doExecute(
  player: Player,
  toolName: string,
  params: Record<string, unknown>,
): Promise<void> {
  const session = this.getSession(player);
  const ctx = new ToolContextImpl({ activeBotName: session.activeBot });

  try {
    const result = await this.toolManager.executeTool(toolName, params, ctx);
    ToolResultRenderer.render(player, toolName, result);
    this.report.append({
      id: `${Date.now()}_${toolName}`,
      timestamp: Date.now(),
      player: player.realName,
      tool: toolName,
      params,
      success: result.success,
      durationMs: result.duration_ms,
      error: result.error,
    });
  } catch (err) {
    ToolResultRenderer.renderError(player, String(err));
  }
}
```

```typescript
function registerCommands(BotManager: any, BotTestSuite: any): void {
  try {
    const cmd = mc.newCommand('mcagent', 'McAgent 插件控制', 0, 0x80);
    cmd.setEnum('TestAction', ['test', 'legacytest']);
    cmd.mandatory('action', 5, 'TestAction', 'TestAction', 1);
    cmd.overload(['TestAction']);
    cmd.overload([]);

    // 实例化 GUI 测试器（从配置读取 enabled / require_op 等）
    const guiTester = new InGameToolTester(toolManager, ConfigManager.getGuiTestConfig());

    cmd.setCallback((_cmd, _ori, out, res) => {
      try {
        const action = res.action ? String(res.action).toLowerCase() : 'info';

        if (action === 'test') {
          if (_ori.player) {
            guiTester.onCommand(_ori.player);
          } else {
            out.error('GUI 模式仅限玩家使用');
          }
          return;
        }

        if (action === 'legacytest' && _ori.player && BotTestSuite) {
          BotTestSuite.showMainMenu(_ori.player);
          return;
        }

        // info 分支保持不变 ...
      } catch (e) {
        out.error('指令执行出错');
        logger.error('指令回调异常', e);
      }
    });

    cmd.setup();
  } catch (e) {
    logger.error('命令注册异常', e);
  }
}
```

## 第7章 风险与应对

| 风险 | 概率 | 影响 | 应对措施 |
|:----:|:----:|:----:|----------|
| `ToolContextImpl` 在 LLSE 运行时不兼容 | 低 | 高 | 提前在 BDS 内验证构造方式；必要时自定义 `PlayerAccess` |
| `BetterForms.js` 路径/模块格式与 TS 编译冲突 | 中 | 中 | 使用 CommonJS 兼容导入或复制到 src/test/ 下重命名 |
| 部分工具 schema 含复杂嵌套对象 | 中 | 低 | 对 object 类型做降级处理：生成 JSON 文本输入框 |
| 旧版 `BotTestSuite` 被移除后影响现有测试习惯 | 低 | 低 | 保留 `/mcagent legacytest` 入口或主菜单「旧版测试套件」按钮 |
| 表单控件过多导致移动端显示异常 | 中 | 中 | 控制单次表单控件数 ≤ 20；过多时分页或分组 |
| 冒烟测试破坏测试服环境 | 中 | 中 | 默认用例使用小范围坐标；执行前再次弹窗确认 |
