## 第3章 关键夹具（Fixture）实现要点

### 3.1 AcMinimalServer（AC 最小化 TCP 服务器）

```typescript
// packages/agent-core/__tests__/it/fixtures/ac-minimal-server.ts
import { TcpServer, ServerEvent } from '../../../src/main/tcp/tcp-server'
import { WorkspaceManager } from '../../../src/main/workspace/workspace-manager'
import { DefaultToolDispatcher } from '../../../src/main/pipeline/tool-dispatcher'
import { normalizeToolSchema } from '../../../src/main/tool-schema'

export interface AcMinimalContext {
  tcpServer: TcpServer
  workspaceManager: WorkspaceManager
  toolDispatcher: DefaultToolDispatcher
  port: number
  authToken: string
  stop: () => Promise<void>
}

export async function startAcMinimalServer(
  port = 27541,
  authToken = 'mct_64cf4ca6c0c64a75aaf9a5b0',
): Promise<AcMinimalContext> {
  const workspaceManager = new WorkspaceManager(false)

  const tcpServer = new TcpServer({
    host: '127.0.0.1',
    port,
    authTokens: new Set([authToken]),
    maxConnections: 10,
  })

  tcpServer.setMessageHandlerFactory((_connId) => ({
    onNotification: (_cid, notif) => {
      if (notif.method === 'register_tools') {
        const params = notif.params as { tools?: unknown[] } | undefined
        if (params?.tools) {
          const normalized = params.tools.map((raw) =>
            normalizeToolSchema(raw as Record<string, unknown>)
          )
          const workspace = workspaceManager.getWorkspaceByConnectionId(_connId)
          if (workspace) {
            workspaceManager.registerTools(workspace.id, normalized)
            console.log(`[AC] 工具注册: ${normalized.length} 个工具`)
          }
        }
      }
      if (notif.method === 'status_report') {
        // 忽略状态上报
      }
    },
    onRequest: async (_cid, req) => {
      if (req.method === 'handshake' || req.method === 'pong') return null
      return {
        jsonrpc: '2.0',
        id: req.id,
        error: { code: -32601, message: `Method not found: ${req.method}` },
      }
    },
  }))

  // 监听连接事件，自动创建 workspace
  tcpServer.on(ServerEvent.ConnectionStateChange, ({ clientId, newState }) => {
    if (newState === 'connected') {
      const conn = tcpServer.getConnection(clientId)
      if (conn?.instanceId) {
        workspaceManager.setOnline(conn.instanceId, clientId)
        console.log(`[AC] 实例 ${conn.instanceId} 已连接`)
      }
    }
  })

  await tcpServer.start()
  const actualPort = (tcpServer as any).server?.address()?.port ?? port

  const toolDispatcher = new DefaultToolDispatcher(workspaceManager, tcpServer)

  return {
    tcpServer,
    workspaceManager,
    toolDispatcher,
    port: actualPort,
    authToken,
    stop: async () => { await tcpServer.stop() },
  }
}
```

**关键设计说明**：
- 不启动 Electron 窗口，仅初始化 TCP 服务 + 工作区管理 + 工具调度
- `normalizeToolSchema` 从独立模块 `tool-schema.ts` 导入，避免触发 Electron 主进程初始化
- 通过 `ServerEvent.ConnectionStateChange` 事件自动同步 TCP 连接状态到 Workspace 在线状态
- 固定端口 27541，与 JE 端 Alice Mod 配置对齐

```typescript
// packages/agent-core/__tests__/it/level2/je-e2e-test.test.ts
import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { spawn } from 'node:child_process'
import path from 'node:path'
import fs from 'node:fs'
import { startAcMinimalServer, type AcMinimalContext } from '../fixtures/ac-minimal-server'

/** 等待 MC 服务端就绪（通过 stdout 检测 "Done" 关键词） */
async function waitForMcReady(
  childProcess: ReturnType<typeof spawn>,
  timeoutMs = 180_000,
): Promise<void> {
  return new Promise<void>((resolve, reject) => {
    const timeout = setTimeout(() => {
      reject(new Error(`MC 服务端启动超时 (${timeoutMs}ms)`))
    }, timeoutMs)
    let output = ''
    const onData = (chunk: Buffer) => {
      const text = chunk.toString()
      output += text
      if (output.includes('Done') || text.includes('Done')) {
        clearTimeout(timeout)
        if (childProcess.stdout) childProcess.stdout.removeListener('data', onData)
        if (childProcess.stderr) childProcess.stderr.removeListener('data', onData)
        resolve()
      }
    }
    if (childProcess.stdout) childProcess.stdout.on('data', onData)
    if (childProcess.stderr) childProcess.stderr.on('data', onData)
    childProcess.on('error', (err) => { clearTimeout(timeout); reject(err) })
    childProcess.on('exit', (code) => {
      clearTimeout(timeout)
      if (code !== 0) reject(new Error(`MC 服务端异常退出，code=${code}`))
    })
  })
}

describe.skipIf(!hasJava21())('L2 端到端: JE 工具注册与调用', () => {

    // 2. 启动 Minecraft 服务端（确保 AC 已就绪，JE 首次连接即成功）
    const serverJavaDir = path.resolve(__dirname, '../../../../../serverjava')

      cwd: serverJavaDir,
      stdio: ['ignore', 'pipe', 'pipe'],

    // 3. 等待 MC 服务端完成初始化

    // 4. 等待 JE 连接并注册工具

    expect(spawnData).toBeDefined()
    expect(spawnData.name).toBe('AliceBot_Move')

    const moveData = await ac.toolDispatcher.callTool(workspaceId, 'move_to', {
      x: 100, y: 70, z: -50,
    }, 15000) as any
    expect(moveData).toBeDefined()
  })

  it('L2-12: 错误路径 - 不存在的假人返回错误', async () => {
    await expect(
      ac.toolDispatcher.callTool(workspaceId, 'bot_info', { name: 'NonExistBot' }, 5000)
    ).rejects.toThrow()
  })
})
```

**设计说明**：
- 使用 `startAcMinimalServer` 启动 in-process AC 服务器，避免 Electron 窗口依赖
- MC 服务端通过 `spawn` 子进程启动，使用 `waitForMcReady` 检测 "Done" 关键词确认就绪
- 测试顺序：先启动 AC → 再启动 MC → 等待 JE 连接并注册工具 → 执行工具调用用例
- 清理阶段：先销毁假人 → 停止 MC 服务端 → 停止 AC 服务器
- 环境检查：`hasJava21()` 检查 JDK 21 可用性，未安装时跳过整个测试套件

## 第6章 CI 接入

### 6.1 GitHub Actions 工作流

```yaml
# .github/workflows/integration-test.yml
name: Integration Test

on:
  workflow_dispatch:
  schedule:
    - cron: '0 2 * * *'    # 每日 2:00 UTC
  push:
    branches: [main]
    paths:
      - 'packages/agent-core/src/**'
      - 'packages/adapter-java/src/**'
      - 'packages/agent-core/__tests__/pipeline/**'
      - 'packages/agent-core/__tests__/tcp/**'
      - 'packages/agent-core/__tests__/it/**'

jobs:
  regression:
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: 20 }
      - run: pnpm install --frozen-lockfile
      - run: pnpm -r build
      - run: pnpm test:integration:regression

  e2e-real-mc:
    runs-on: ubuntu-latest
    timeout-minutes: 30
    needs: regression
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: 20 }
      - uses: actions/setup-java@v4
        with: { distribution: temurin, java-version: 21 }
      - run: pnpm install --frozen-lockfile
      - run: pnpm -r build
      - name: 准备 MC 服务端
        run: |
          mkdir -p serverjava
          cp server-jar-paths/server.jar serverjava/
      - name: 拷贝 Alice Mod
        run: cp packages/adapter-java/build/libs/alice-mod-*.jar serverjava/mods/
      - name: 跑 L2 集成测试
        run: pnpm test:integration:l2
      - name: 上传日志
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: mc-logs-${{ github.run_id }}
          path: serverjava/logs/
          retention-days: 14
```

### 6.2 Job 触发策略

| Job | 触发 | 阻塞 PR | 时长 |
|-----|------|---------|------|
| `regression` | push + schedule | 是 | < 5s |
| `e2e-real-mc` | push (仅 main) + schedule | 否 | 3-5 分钟 |

PR 门禁只跑回归层。`e2e-real-mc` 在 main 分支 push 与 nightly 跑。

### 6.3 失败快速定位

- 回归层失败：直接读 vitest 报错，对应 Bug 编号
- L2 失败：检查 vitest 输出 + MC 服务器日志（`serverjava/logs/latest.log`）
- MC 服务器崩溃：检查 `serverjava/logs/` 中 stacktrace
- TCP 连接失败：检查 AC 服务器是否已启动，端口是否被占用
