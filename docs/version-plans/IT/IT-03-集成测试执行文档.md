```typescript
// packages/agent-core/__tests__/it/fixtures/mock-je-client.ts
import { createConnection, Socket } from 'node:net'
import { JsonRpcCodec } from '../../../../src/main/tcp/json-rpc'

export class MockJeClient {
  private socket!: Socket
  private buffer = ''
  private requestId = 0
  private pendingRequests = new Map<number, (resp: any) => void>()
  private onRequest: ((req: any) => Promise<any>) | null = null

  public sentRequests: any[] = []
  public sentNotifications: any[] = []
  public receivedRequests: any[] = []

  constructor(private host: string, private port: number) {}

  async connect(): Promise<void> {
    return new Promise((resolve, reject) => {
      this.socket = createConnection(this.port, this.host, () => resolve())
      this.socket.on('data', (d: Buffer) => this.handleData(d))
      this.socket.on('error', reject)
    })
  }

  async handshake(params: any): Promise<any> {
    return this.sendRequest('handshake', params)
  }

  sendRegisterTools(tools: any[]): void {
    this.sendNotification('register_tools', { tools })
  }

  private sendRequest(method: string, params: any): Promise<any> {
    const id = ++this.requestId
    const req = { jsonrpc: '2.0', id, method, params }
    this.sentRequests.push(req)
    return new Promise((resolve) => {
      this.pendingRequests.set(id, resolve)
      this.socket.write(JsonRpcCodec.encodeRequest(req) + '\n')
    })
  }

  private sendNotification(method: string, params: any): void {
    const notif = { jsonrpc: '2.0', method, params }
    this.sentNotifications.push(notif)
    this.socket.write(JsonRpcCodec.encodeNotification(notif) + '\n')
  }

  private handleData(data: Buffer): void {
    this.buffer += data.toString()
    const lines = this.buffer.split('\n')
    this.buffer = lines.pop() ?? ''
    for (const line of lines) {
      if (!line.trim()) continue
      const msg = JSON.parse(line)
      if (msg.id !== undefined && msg.method) {
        this.receivedRequests.push(msg)
        this.handleIncomingRequest(msg)
      } else if (msg.id !== undefined && (msg.result !== undefined || msg.error !== undefined)) {
        const resolve = this.pendingRequests.get(msg.id)
        resolve?.(msg)
        this.pendingRequests.delete(msg.id)
      }
    }
  }

  private async handleIncomingRequest(req: any): Promise<void> {
    if (this.onRequest) {
      const response = await this.onRequest(req)
      this.socket.write(JsonRpcCodec.encodeResponse({ id: req.id, result: response }) + '\n')
    }
  }

  setOnRequestHandler(handler: (req: any) => Promise<any>): void {
    this.onRequest = handler
  }

  disconnect(): void {
    this.socket?.destroy()
  }
}
```

```typescript
// packages/agent-core/__tests__/it/helpers/ac-bootstrap.ts
import { TcpServer } from '../../../src/main/tcp/tcp-server'
import { WorkspaceManager } from '../../../src/main/workspace/workspace-manager'
import { DefaultToolDispatcher } from '../../../src/main/pipeline/tool-dispatcher'
import { PromptBuilder } from '../../../src/main/prompt/builder/prompt-builder'
import { handleTcpNotification } from '../../../src/main/index'
import { getLogger } from '../../../src/main/log'

export interface AcItContext {
  tcpServer: TcpServer
  workspaceManager: WorkspaceManager
  promptBuilder: PromptBuilder
  toolDispatcher: DefaultToolDispatcher
  port: number
  authToken: string
  stop: () => Promise<void>
}

export async function bootstrapAcForIT(): Promise<AcItContext> {
  process.env.ALICE_IT_MODE = '1'
  process.env.ALICE_IT_USER_DATA = process.env.ALICE_IT_USER_DATA
    ?? `${process.cwd()}/.it-runtime/userdata-${process.pid}`

  const workspaceManager = new WorkspaceManager(false)
  const authToken = process.env.ALICE_AUTH_TOKEN ?? 'it-test-token'
  const tcpServer = new TcpServer({
    host: '127.0.0.1',
    port: 0,
    authTokens: new Set([authToken]),
    maxConnections: 10,
  })

  const logger = getLogger()
  tcpServer.setMessageHandlerFactory((_connId) => ({
    onNotification: (_cid, notif) => handleTcpNotification(workspaceManager, notif, _cid, logger),
    onRequest: async (_cid, req) => {
      if (req.method === 'handshake' || req.method === 'pong') return null
      return {
        jsonrpc: '2.0',
        id: req.id,
        error: { code: -32601, message: `Method not found: ${req.method}` },
      }
    },
  }))

  await tcpServer.start()
  const port = (tcpServer as any).server.address().port

  const promptBuilder = new PromptBuilder({
    toolRegistry: workspaceManager.getToolRegistry(),
  })
  const toolDispatcher = new DefaultToolDispatcher(workspaceManager, tcpServer)

  return {
    tcpServer,
    workspaceManager,
    promptBuilder,
    toolDispatcher,
    port,
    authToken,
    stop: async () => { await tcpServer.stop() },
  }
}
```

```java
        // 等待 JE TcpClient 连上 AC（通过日志关键字）

    @AfterAll

    private Path findMod(Path modsDir, String prefix) throws Exception {
        try (var stream = Files.list(modsDir)) {
            return stream.filter(p -> p.getFileName().toString().contains(prefix))
                .findFirst()
                .orElseThrow(() -> new IllegalStateException("Mod not found: " + prefix));
        }
    }

    protected void waitForMcLog(String regex, long timeoutMs) throws Exception {
        // 读取 latest.log 等待匹配
        long deadline = System.currentTimeMillis() + timeoutMs;
        while (System.currentTimeMillis() < deadline) {
            if (Files.exists(mcServer.getLogFile())) {
                List<String> lines = Files.readAllLines(mcServer.getLogFile());
                for (String line : lines) {
                    if (line.matches(".*" + regex + ".*")) return;
                }
            }
            Thread.sleep(500);
        }
        throw new TimeoutException("Timed out waiting for log: " + regex);
    }
}
```

#### 4.1.2 B4 — AC normalizeToolSchema

由于 `normalizeToolSchema` 在 `index.ts` 中是模块内函数，回归测试需要先把它 export。修复方案：在 `index.ts` 末尾加 `export { normalizeToolSchema }`。

```typescript
// packages/agent-core/__tests__/tcp/register-tools-normalize.test.ts
import { describe, it, expect } from 'vitest'
import { normalizeToolSchema } from '../../../src/main/index'

describe('B4 回归: normalizeToolSchema 兼容 JE input_schema', () => {
  it('JE 格式 input_schema 应转换为 parameters（type/description/required）', () => {
    const jeRaw = {
      name: 'bot_spawn',
      description: '创建假人',
      category: '',
      input_schema: {
        type: 'object',
        properties: {
          name: { type: 'string', description: '假人名' },
          x: { type: 'number', description: 'X' },
          y: { type: 'number', description: 'Y' },
          z: { type: 'number', description: 'Z' },
        },
        required: ['name', 'x', 'y', 'z'],
      },
    }
    const r = normalizeToolSchema(jeRaw)
    expect(r.name).toBe('bot_spawn')
    expect(r.parameters).toBeDefined()
    expect(r.parameters.name.type).toBe('string')
    expect(r.parameters.name.required).toBe(true)
    expect(r.parameters.x.required).toBe(true)
    expect(r.parameters.y.required).toBe(true)
    expect(r.parameters.z.required).toBe(true)
    expect(r.category).toBe('entity')  // 按 bot_* 前缀推断
  })

  it('BE 格式 parameters 应直接保留', () => {
    const beRaw = {
      name: 'move_to',
      description: '移动',
      category: 'movement',
      parameters: {
        x: { type: 'number', description: 'X', required: true },
        y: { type: 'number', description: 'Y', required: true },
        z: { type: 'number', description: 'Z', required: true },
      },
    }
    const r = normalizeToolSchema(beRaw)
    expect(r.parameters.x.required).toBe(true)
    expect(r.category).toBe('movement')
  })

  it('category 缺失时按工具名前缀推断', () => {
    expect(normalizeToolSchema({ name: 'move_to', description: '' }).category).toBe('movement')
```

```java
     * 直接构造 WorldContext 的最小子集。
     * 因为 registerTools / handleToolCall 是 private 反射访问。

    @Test
    void testRegisterToolsIsNotification() throws Exception {

            public String name() { return "bot_spawn"; }

            }
        };
        ToolRegistry.register(stubTool);

        // 模拟 AC 发来 tool_call Request
        Map<String, Object> args = Map.of("name", "AliceBot", "x", 10.0, "y", 64.0, "z", 20.0);

        params.add("parameters", JsonParser.parseString(new Gson().toJson(args)));

        JsonRpcMessage.Request req = new JsonRpcMessage.Request(
            JsonRpcId.of(1), "tool_call", params);

        // 反射调用 handleToolCall
        WorldContext ctx = WorldContextBuilder.minimal(new MockTcpClient());
        AtomicReference<JsonElement> response = new AtomicReference<>();
        var m = WorldContext.class.getDeclaredMethod("handleToolCall",
            JsonRpcMessage.Request.class, TcpClient.BiConsumer.class);
        m.setAccessible(true);
        m.invoke(ctx, req, (id, r) -> response.set(r));
```
