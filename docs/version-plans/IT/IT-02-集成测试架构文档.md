│  │  Level 2: 端到端（AC In-process + 真实 JE + 真实 MC，3-5 分钟）   │  │

```typescript

import { describe, it, expect, vi, beforeEach } from 'vitest'
import { DefaultToolDispatcher } from '../../../src/main/pipeline/tool-dispatcher'
import { TcpServer } from '../../../src/main/tcp/tcp-server'
import { WorkspaceManager } from '../../../src/main/workspace/workspace-manager'

describe('B1: ToolDispatcher.callTool 必须发 method=tool_call', () => {
  it('TCP 层收到的 method 字段是 tool_call，不是 call_tool', async () => {
    const wm = new WorkspaceManager(false)
    const server = new TcpServer({ host: '127.0.0.1', port: 0 })
    await server.start()
    const port = (server as any).server.address().port
    const dispatcher = new DefaultToolDispatcher(wm, server)

    // 录制所有发往连接的 JSON
    const sentJson: any[] = []
    server.on('connection:new', (conn: any) => {
      const original = conn.sendJson.bind(conn)
      conn.sendJson = (msg: any) => { sentJson.push(msg); return original(msg) }
    })

    // 准备一个 online workspace
    wm.createWorkspace({ instanceId: 'it-b1' })
    wm.setOnline('it-b1', 'fake-conn-id')

    try {
      await expect(
        dispatcher.callTool('fake-conn-id', 'bot_spawn', { name: 'X', x: 0, y: 64, z: 0 })
      ).rejects.toBeTruthy()  // 期望失败因为无真实 connection，但 JSON 必须先发出

      const toolCallReq = sentJson.find((m) => m.method?.startsWith('tool') || m.method?.startsWith('call_tool'))
      expect(toolCallReq).toBeDefined()
      expect(toolCallReq.method).toBe('tool_call')
      expect(toolCallReq.method).not.toBe('call_tool')
    } finally {
      await server.stop()
    }
  })
})
```

#### 3.2.1 AcMinimalServer（AC 最小化 TCP 服务器）

```typescript
// packages/agent-core/__tests__/it/fixtures/ac-minimal-server.ts
import { TcpServer, ServerEvent } from '../../../src/main/tcp/tcp-server'
import { WorkspaceManager } from '../../../src/main/workspace/workspace-manager'
import { DefaultToolDispatcher } from '../../../src/main/pipeline/tool-dispatcher'
import { normalizeToolSchema } from '../../../src/main/tool-schema'

export async function startAcMinimalServer(
  port = 27541,
  authToken = 'mct_64cf4ca6c0c64a75aaf9a5b0',
): Promise<AcMinimalContext> {
  const workspaceManager = new WorkspaceManager(false)
  const tcpServer = new TcpServer({
    host: '127.0.0.1', port,
    authTokens: new Set([authToken]), maxConnections: 10,
  })
  tcpServer.setMessageHandlerFactory((_connId) => ({
    onNotification: (_cid, notif) => {
      if (notif.method === 'register_tools') {
        const params = notif.params as { tools?: unknown[] } | undefined
        if (params?.tools) {
          const normalized = params.tools.map((raw) =>
            normalizeToolSchema(raw as Record<string, unknown>)
          )
          const workspace = workspaceManager.getWorkspaceByConnectionId(_connId)
          if (workspace) workspaceManager.registerTools(workspace.id, normalized)
        }
      }
    },
    onRequest: async (_cid, req) => {
      if (req.method === 'handshake' || req.method === 'pong') return null
      return { jsonrpc: '2.0', id: req.id, error: { code: -32601, message: `Method not found` } }
    },
  }))
  tcpServer.on(ServerEvent.ConnectionStateChange, ({ clientId, newState }) => {
    if (newState === 'connected') {
      const conn = tcpServer.getConnection(clientId)
      if (conn?.instanceId) workspaceManager.setOnline(conn.instanceId, clientId)
    }
  })
  await tcpServer.start()
  const toolDispatcher = new DefaultToolDispatcher(workspaceManager, tcpServer)
  return { tcpServer, workspaceManager, toolDispatcher, port: actualPort, authToken,
    stop: async () => { await tcpServer.stop() } }
}
```

**设计要点**：
- 不启动 Electron 窗口，仅初始化 TCP 服务 + 工作区管理 + 工具调度
- `normalizeToolSchema` 从独立模块 `tool-schema.ts` 导入，避免触发 Electron 主进程初始化
- 通过 `ServerEvent.ConnectionStateChange` 事件自动同步 TCP 连接状态到 Workspace 在线状态

## 第5章 关键技术决策

### 5.1 回归层独立于 IT

- 原因：回归测试是契约基线，应在 PR 门禁时跑（每次提交），不能依赖 MC 服务器
- 实现：放在 `__tests__/pipeline/`、`__tests__/tcp/` 路径下，与单元测试同 runner
- 与 IT 的关系：回归测试必须先于 IT 通过（PR 阶段）；IT 验证完整业务流（nightly / release）

### 5.2 归一化放在 AC 端而非 JE 端

- 原因：协议消费方应做容错，发送方应该被强制遵循协议（但允许 B4 旧代码继续运行）
- 折中：JE 端 SchemaGenerator 暂不改动（避免大范围修改），但新增 warning 日志引导未来修复
- 实际：B4 修复采用"消费方归一化"策略

### 5.3 L1 协议契约测试未实现

- 原因：回归层已覆盖关键协议字段验证（B1 method 名、B4 Schema 归一化），L1 额外价值主要体现在协议流编排验证，可通过 L2 端到端测试间接覆盖
- 风险：mock 客户端测试可更早捕获协议格式问题，未来可按需补

### 5.4 AC In-Process 启动

- 原因：避免子进程管理与 IPC 复杂度
- 实现：通过 `startAcMinimalServer` 启动，跳过 Electron 窗口
- 风险：与 Electron 主进程略有差异（如 `app.getPath('userData')`）
- 缓解：在 IT 模式下使用固定端口和临时目录

### 5.5 测试隔离

- 端口隔离：AC 启动在 `27541`，MC 在 `25565`
- 数据库隔离：不使用持久化数据库，仅内存操作
- 进程隔离：AC 为 in-process，MC 为 spawn 子进程

### 5.6 时间预算

| 阶段 | 时间 |
|------|------|
| 回归层（vitest unit） | < 1s |
| L2 MC 启动 | 60-120s |
| L2 用例执行 | 10-30s |
| L2 清理 | 5-10s |
| **总时长** | **2-3 分钟** |

## 第8章 总结

集成测试架构的核心是 **Bug 修复 + 真实端到端验证**：

- **回归层**（< 1s）：固化 B1/B4 修复，PR 门禁基线，确保协议字段不被未来重构破坏
- **L2 端到端**（2-3 分钟）：真实 JE 模组 + 真实 MC 服务器 + 真实假人，验证 AC→JE→MC 完整链路

完成交付：**4 个 Bug 修复 + 7 个测试用例**，覆盖工具注册、假人创建、假人查询、假人移动、错误路径等核心场景，BE 端不在范围，专注 AC↔JE。
