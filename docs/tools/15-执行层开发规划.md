#### Step 1.2: 通用移动工具（move_to）

**任务**：实现 move_to，通过 target_type 支持坐标/实体/方块三种目标

**实现要点**：
- 调用寻路系统
- 通过 `target_type` 参数区分目标类型：`coordinate` | `entity` | `block`
- 实体目标自动追踪移动，需停在指定距离外
- 支持停止距离（distance）和疾跑（sprint）参数
- 合并原 move_to_entity、move_to_block 功能

**代码结构**：
```typescript
interface MoveToParams {
  target: { x: number, y: number, z: number } | { entity_id: string };
  target_type: 'coordinate' | 'entity' | 'block';
  distance?: number;       // 停止距离（默认2格，实体目标默认3格）
  sprint?: boolean;        // 是否疾跑
}

async function move_to(params: MoveToParams): Promise<MoveResult> {
  // 1. 解析目标位置
  let targetPos: Vec3;
  if (params.target_type === 'coordinate' || params.target_type === 'block') {
    targetPos = new Vec3(params.target.x, params.target.y, params.target.z);
  } else {
    const entity = await world.getEntity(params.target.entity_id);
    if (!entity) {
      return { success: false, error: '实体不存在' };
    }
    targetPos = entity.position;
  }
  
  // 2. 计算停止位置
  const stopDistance = params.distance || (params.target_type === 'entity' ? 3 : 2);
  const current = await player.getPosition();
  const direction = targetPos.subtract(current).normalize();
  const stopPos = targetPos.subtract(direction.multiply(stopDistance));
  
  // 3. 寻路
  const path = await pathfinder.findPath(current, stopPos, {
    avoid_danger: true,
    max_jump: 3,
    allow_swim: true
  });
  
  // 4. 执行路径
  const result = await executionAI.executePath(path, { sprint: params.sprint });
  
  return {
    success: result.success,
    from: current,
    to: result.final_position,
    distance: result.distance,
    time_cost: result.time_cost,
    hunger_cost: result.hunger_cost,
    error: result.error
  };
}
```

```typescript
async function ride(params: { entity_id: string }): Promise<RideResult> {
  const entity = await world.getEntity(params.entity_id);
  if (!entity) {
    return { success: false, error: '实体不存在' };
  }
  
  // 检查是否可骑乘

    return { success: false, error: '该实体不可骑乘' };

  // 执行骑乘
  const success = await executionAI.ride(entity);
  
  return {
    success,
    entity_id: params.entity_id,
    entity_type: entity.type,
    error: success ? undefined : '骑乘失败'
  };
}
```

#### Step 2.1: 丢弃物品

**任务**：实现 drop_item，支持丢弃到地上或交给实体

**实现要点**：
- 支持指定物品名/数量（不指定count丢全部）
- 支持向指定实体丢弃（如与村民交互时）
- 返回丢弃结果

**代码结构**：
```typescript
async function drop_item(params: { 
  item_name?: string, 
  count?: number, 
  target_entity?: string 
}): Promise<DropResult> {
  const inventory = await player.getInventory();
  const itemSlot = params.item_name ? inventory.findItem(params.item_name) : inventory.selectedSlot;
  
  if (!itemSlot) {
    return { success: false, error: '物品不存在' };
  }
  
  const count = params.count || itemSlot.count;
  const actualCount = Math.min(count, itemSlot.count);
  
  // 执行丢弃（向实体或地上）
  if (params.target_entity) {
    await executionAI.dropToEntity(itemSlot, actualCount, params.target_entity);
  } else {
    await executionAI.dropItem(itemSlot, actualCount);
  }
  
  return {
    success: true,
    item: itemSlot.name,
    count: actualCount,
    remaining: itemSlot.count - actualCount,
    target: params.target_entity || 'ground'
  };
}
```

```typescript
async function equip_item(params: { 
  item_name: string, 
  slot: 'head' | 'chest' | 'legs' | 'feet' | 'offhand',
  action: 'equip' | 'unequip'
}): Promise<EquipResult> {
  if (params.action === 'equip') {
    const inventory = await player.getInventory();
    const item = inventory.findItem(params.item_name);
    if (!item) {
      return { success: false, error: '背包中没有该物品' };

    return {
      success,
      slot: params.slot,
      action: 'unequip',
      error: success ? undefined : '卸下失败'
    };
  }
}
```

**代码结构**：
```typescript
async function sleep(params: { 
  action: 'sleep' | 'wake',
  bed_pos?: { x: number, y: number, z: number },
  wait_seconds?: number
}): Promise<SleepResult> {
  if (params.action === 'wake') {
    await executionAI.wakeUp();
    return { success: true, action: 'wake' };
  }
  
  // 1. 找床
  let bed;
  if (params.bed_pos) {
    bed = await world.getBlock(params.bed_pos);
  } else {
    bed = await findNearestBed();
  }
  
  if (!bed) {
    return { success: false, error: '附近没有床' };
  }
  
  // 2. 移动到床附近
  await move_to({ target: { x: bed.x, y: bed.y, z: bed.z }, target_type: 'block', distance: 1 });
  
  // 3. 执行睡觉
  await executionAI.sleep(bed);
  
  // 4. 等待天亮或指定时间
  const waitTime = params.wait_seconds || 0;
  if (waitTime > 0) {
    await sleep(waitTime * 1000);
  } else {
    await waitForDay();
  }
  
  return {
    success: true,
    slept_duration: Date.now() - startTime,
    time_when_wake: await world.getTime()
  };
}
```

#### Step 4.2: 区域操作

**实现要点**：

```typescript
async function area_operation(params: {
  mode: 'fill' | 'clear' | 'break' | 'vein',
  from: { x: number, y: number, z: number },
  to?: { x: number, y: number, z: number },
  block_name?: string,     // fill 模式需要
  radius?: number          // vein 模式需要，默认16
}): Promise<AreaResult> {
  switch (params.mode) {
    case 'fill':
      return fillArea(params.from, params.to!, params.block_name!);
    case 'clear':
      return clearArea(params.from, params.to!);
    case 'break':
      return breakBlocks(params.from, params.to!);
    case 'vein':
      return mineVein(params.from, params.radius || 16);
  }
}

// 填充区域
async function fillArea(from: Vec3, to: Vec3, block: string): Promise<AreaResult> {
  const blocks = getBlocksInArea(from, to);
  let successCount = 0;
  let failCount = 0;
  
  for (const pos of blocks) {
    try {
      await place_block({ x: pos.x, y: pos.y, z: pos.z, block_name: block });
      successCount++;
    } catch (e) {
      failCount++;
    }
  }
  
  return { success: failCount === 0, total: blocks.length, success_count: successCount, fail_count: failCount };
}

// 连锁挖掘矿脉
async function mineVein(start: Vec3, radius: number): Promise<VeinResult> {
  const block = await world.getBlock(start);
  if (!isOre(block.type)) {
    return { success: false, error: '不是矿石' };
  }
  
  const veinBlocks = await findConnectedOres(start, block.type, radius);
  let minedCount = 0;
  for (const pos of veinBlocks) {
    await mine_block({ x: pos.x, y: pos.y, z: pos.z });
    minedCount++;
  }
  
  return { success: true, block_type: block.type, mined_count: minedCount, total_ores: veinBlocks.length };
}
```

#### Step 5.2: 通用生物交互

**任务**：实现 interact_entity，通过 action 参数支持 6 种交互

**实现要点**：
- 移动到实体附近
- 通过 `action` 参数区分：feed/breed/trade/tame/shear/milk
- 每种交互的参数不同（trade 需要 trade_index）
- 合并原 feed_entity、breed_entity、trade_with_villager、tame_entity、shear_entity、milk_entity

**代码结构**：
```typescript
async function interact_entity(params: {
  entity_id: string,
  action: 'feed' | 'breed' | 'trade' | 'tame' | 'shear' | 'milk',
  item_name?: string,     // feed/breed/tame 需要
  trade_index?: number,   // trade 模式需要
  count?: number          // trade 交易数量
}): Promise<InteractResult> {
  const entity = await world.getEntity(params.entity_id);
  if (!entity) {
    return { success: false, error: '实体不存在' };
  }
  
  // 移动到实体附近
  await move_to({ target: { entity_id: params.entity_id }, target_type: 'entity', distance: 2 });
  
  // 根据 action 执行不同交互
  let success: boolean;
  switch (params.action) {
    case 'feed': {
      if (!isEdibleBy(entity.type, params.item_name!)) {
        return { success: false, error: '该生物不吃这个' };
      }
      success = await executionAI.feedEntity(entity, params.item_name!);
      break;
    }
    case 'breed':
      success = await executionAI.breedEntity(entity, params.item_name!);
      break;
    case 'trade': {
      const trades = await entity.getTrades();
      if (!trades[params.trade_index!]) {
        return { success: false, error: '交易不存在' };
      }
      success = await executionAI.trade(entity, params.trade_index!, params.count || 1);
      break;
    }
    case 'tame':
      success = await executionAI.tameEntity(entity, params.item_name!);
      break;
    case 'shear':
      success = await executionAI.shearEntity(entity);
      break;
    case 'milk':
      success = await executionAI.milkEntity(entity);
      break;
  }
  
  return {
    success,
    entity_id: params.entity_id,
    entity_type: entity.type,
    action: params.action,
    error: success ? undefined : `${params.action} 失败`
  };
}
```

#### Step 5.3: 牵引系统

**任务**：实现 lead_entity（通过 action 参数区分牵引/释放）

**实现要点**：
- 移动到实体附近
- 执行牵引/释放动作
- 合并原 lead_entity、release_entity

**代码结构**：
```typescript
async function lead_entity(params: {
  entity_id: string,
  action: 'lead' | 'release'
}): Promise<LeadResult> {
  const entity = await world.getEntity(params.entity_id);
  if (!entity) {
    return { success: false, error: '实体不存在' };
  }
  
  if (params.action === 'lead') {
    // 移动到实体附近
    await move_to({ target: { entity_id: params.entity_id }, target_type: 'entity', distance: 1 });
    const success = await executionAI.leadEntity(entity);
    return { success, entity_id: params.entity_id, action: 'lead', error: success ? undefined : '牵引失败' };
  } else {
    const success = await executionAI.releaseEntity(entity);
    return { success, entity_id: params.entity_id, action: 'release', error: success ? undefined : '释放失败' };
  }
}
```
