# 方块与物品工具设计（Block & Item Tools Design）

> 版本：v1.0
> 日期：2026-07-01
> 架构：LLM 调用工具 → 执行AI 执行操作 → 返回结果

---

## 0. 设计原则

### 0.1 核心概念

**方块操作由执行AI负责，LLM 只调用工具**

```
┌─────────────────────────────────────────────────────┐
│  LLM（大脑）                                         │
│  - 决定要挖/放/用方块                                 │
│  - 调用方块工具                                       │
└────────────────┬────────────────────────────────────┘
                 │ 调用工具
                 ▼
        ┌────────────────┐
        │  方块工具        │
        │  - mine_block   │
        │  - place_block  │
        │  - use_block    │
        │  - area_operation │
        └────────┬───────┘
                 │ 委托执行
                 ▼
        ┌────────────────┐
        │  执行AI（假人）  │
        │  - 自动选择工具  │
        │  - 自动寻路      │
        │  - 批量操作      │
        └────────┬───────┘
                 │ 返回结果
                 ▼
        ┌────────────────┐
        │  工具返回        │
        │  - 成功/失败    │
        │  - 操作详情     │
        └────────────────┘
```

### 0.2 设计原则

| 原则 | 说明 |
|---|---|
| **自动选择** | 执行AI自动选择合适工具（挖石头用镐、挖土用锹） |
| **批量操作** | 支持区域填充、批量破坏 |
| **有反馈** | 工具返回操作结果（成功/失败/详情） |
| **可失败** | 工具损坏、方块不可破坏、距离太远等 |

### 2.1 mine_block（挖掘方块）

```typescript
interface MineBlockParams {
  x: number;
  y: number;
  z: number;
  options?: {
    tool?: string;  // 指定工具（不填则自动选择）
    silk_touch?: boolean;  // 使用精准采集
  };
}

interface MineBlockResult {
  success: boolean;
  reason?: 'success' | 'block_not_found' | 'not_breakable' | 'tool_broken' | 'too_far' | 'error';
  blockName?: string;
  drops?: Array<{ name: string; count: number }>;
  toolUsed?: string;
  toolDurability?: number;
  errorMessage?: string;
}
```

}

### 3.2 执行AI接口

```typescript
// 执行AI接口（方块操作）
interface ExecutionAI {
  // 单个操作
  mineBlock(x: number, y: number, z: number, options?: MineOptions): Promise<MineBlockResult>;
  placeBlock(x: number, y: number, z: number, blockName: string, facing?: string): Promise<PlaceBlockResult>;
  useBlock(x: number, y: number, z: number): Promise<UseBlockResult>;
  
  // 区域操作
  areaOperation(mode: 'fill' | 'clear' | 'break' | 'vein', from: {x,y,z}, to: {x,y,z}, blockName?: string, radius?: number): Promise<AreaOperationResult>;
}

// 挖掘选项
interface MineOptions {
  tool?: string;
  silk_touch?: boolean;
}
```

### 3.3 执行AI实现（示例）

```typescript
class ExecutionAIImpl implements ExecutionAI {
  private player: Player;
  
  constructor(player: Player) {
    this.player = player;
  }
  
  async mineBlock(x: number, y: number, z: number, options?: MineOptions): Promise<MineBlockResult> {
    // 检查距离
    const distance = this.calculateDistance(this.player.pos, {x, y, z});
    if (distance > 5) {
      return { success: false, reason: 'too_far' };
    }
    
    // 获取方块
    const block = mc.getBlock(x, y, z);
    if (!block) {
      return { success: false, reason: 'block_not_found' };
    }
    
    // 检查是否可破坏
    if (!block.canBeBroken()) {
      return { success: false, reason: 'not_breakable' };
    }
    
    // 选择工具
    const tool = options?.tool || this.selectToolForBlock(block);
    if (!tool) {
      return { success: false, reason: 'no_tool' };
    }
    
    // 装备工具
    const inventory = this.player.getInventory();
    const toolSlot = inventory.findItem(tool);
    if (toolSlot === -1) {
      return { success: false, reason: 'no_tool' };
    }
    
    this.player.setHandSlot(toolSlot);
    
    // 挖掘
    const drops = await this.player.mineBlock(block);
    
    // 获取工具耐久
    const toolItem = inventory.getItem(toolSlot);
    const toolDurability = toolItem.getDurability();
    
    return {
      success: true,
      reason: 'success',
      blockName: block.name,
      drops,
      toolUsed: tool,
      toolDurability
    };
  }
  
  async placeBlock(x: number, y: number, z: number, blockName: string, facing?: string): Promise<PlaceBlockResult> {
    // 检查距离
    const distance = this.calculateDistance(this.player.pos, {x, y, z});
    if (distance > 5) {
      return { success: false, reason: 'too_far' };
    }
    
    // 检查位置是否被占用
    const block = mc.getBlock(x, y, z);
    if (block && !block.isAir()) {
      return { success: false, reason: 'position_occupied' };
    }
    
    // 查找方块
    const inventory = this.player.getInventory();
    const slot = inventory.findItem(blockName);
    if (slot === -1) {
      return { success: false, reason: 'no_block' };
    }
    
    // 装备方块
    this.player.setHandSlot(slot);
    
    // 放置
    const success = await this.player.placeBlock(x, y, z, facing);
    if (!success) {
      return { success: false, reason: 'cannot_place' };
    }
    
    return {
      success: true,
      reason: 'success',
      blockName,
      position: { x, y, z }
    };
  }
  
  async useBlock(x: number, y: number, z: number): Promise<UseBlockResult> {
    // 检查距离
    const distance = this.calculateDistance(this.player.pos, {x, y, z});
    if (distance > 5) {
      return { success: false, reason: 'too_far' };
    }
    
    // 获取方块
    const block = mc.getBlock(x, y, z);
    if (!block) {
      return { success: false, reason: 'block_not_found' };
    }
    
    // 使用方块
    const interactionType = await this.player.useBlock(block);
    
    return {
      success: true,
      reason: 'success',
      blockName: block.name,
      interactionType
    };
  }
  
  async areaOperation(mode: 'fill' | 'clear' | 'break' | 'vein', from: {x,y,z}, to: {x,y,z}, blockName?: string, radius: number = 5): Promise<AreaOperationResult> {
    // 计算区域大小
    const width = Math.abs(to.x - from.x) + 1;
    const height = Math.abs(to.y - from.y) + 1;
    const depth = Math.abs(to.z - from.z) + 1;
    const volume = width * height * depth;
    
    // 检查区域大小
    if (volume > 10000) {
      return { success: false, reason: 'too_large' };
    }
    
    if (mode === 'fill') {
      // 填充区域
      if (!blockName) {
        return { success: false, reason: 'no_block' };
      }
      const inventory = this.player.getInventory();
      const slot = inventory.findItem(blockName);
      if (slot === -1) {
        return { success: false, reason: 'no_block' };
      }
      this.player.setHandSlot(slot);
      
      let blocksPlaced = 0;
      for (let x = Math.min(from.x, to.x); x <= Math.max(from.x, to.x); x++) {
        for (let y = Math.min(from.y, to.y); y <= Math.max(from.y, to.y); y++) {
          for (let z = Math.min(from.z, to.z); z <= Math.max(from.z, to.z); z++) {
            await this.player.placeBlock(x, y, z);
            blocksPlaced++;
          }
        }
      }
      return { success: true, reason: 'success', blocksAffected: blocksPlaced, area: { width, height, depth } };
    }
    
    if (mode === 'clear') {
      // 清空区域
      let blocksCleared = 0;
      for (let x = Math.min(from.x, to.x); x <= Math.max(from.x, to.x); x++) {
        for (let y = Math.min(from.y, to.y); y <= Math.max(from.y, to.y); y++) {
          for (let z = Math.min(from.z, to.z); z <= Math.max(from.z, to.z); z++) {
            const block = mc.getBlock(x, y, z);
            if (block && !block.isAir()) {
              await this.player.breakBlock(block);
              blocksCleared++;
            }
          }
        }
      }
      return { success: true, reason: 'success', blocksAffected: blocksCleared, area: { width, height, depth } };
    }
    
    if (mode === 'break') {
      // 破坏区域内方块
      let blocksBroken = 0;
      const drops: Record<string, number> = {};
      for (let x = Math.min(from.x, to.x); x <= Math.max(from.x, to.x); x++) {
        for (let y = Math.min(from.y, to.y); y <= Math.max(from.y, to.y); y++) {
          for (let z = Math.min(from.z, to.z); z <= Math.max(from.z, to.z); z++) {
            const block = mc.getBlock(x, y, z);
            if (!block || block.isAir()) continue;
            const blockDrops = await this.player.mineBlock(block);
            blocksBroken++;
            for (const drop of blockDrops) {
              drops[drop.name] = (drops[drop.name] || 0) + drop.count;
            }
          }
        }
      }
      return {
        success: true, reason: 'success', blocksAffected: blocksBroken,
        drops: Object.entries(drops).map(([name, count]) => ({ name, count })),
        area: { width, height, depth }
      };
    }
    
    if (mode === 'vein') {
      // 连锁挖掘
      const startBlock = mc.getBlock(from.x, from.y, from.z);
      if (!startBlock || !this.isOre(startBlock)) {
        return { success: false, reason: 'no_ore' };
      }
      
      const oreName = startBlock.name;
      const visited = new Set<string>();
      const queue = [{x: from.x, y: from.y, z: from.z}];
      const oreBlocks: Array<{x,y,z}> = [];
      
      while (queue.length > 0) {
        const pos = queue.shift()!;
        const key = `${pos.x},${pos.y},${pos.z}`;
        if (visited.has(key)) continue;
        visited.add(key);
        
        const block = mc.getBlock(pos.x, pos.y, pos.z);
        if (!block || block.name !== oreName) continue;
        
        const distance = this.calculateDistance(from, pos);
        if (distance > radius) continue;
        
        oreBlocks.push(pos);
        
        const neighbors = [
          {x: pos.x+1, y: pos.y, z: pos.z}, {x: pos.x-1, y: pos.y, z: pos.z},
          {x: pos.x, y: pos.y+1, z: pos.z}, {x: pos.x, y: pos.y-1, z: pos.z},
          {x: pos.x, y: pos.y, z: pos.z+1}, {x: pos.x, y: pos.y, z: pos.z-1}
        ];
        
        for (const neighbor of neighbors) {
          if (!visited.has(`${neighbor.x},${neighbor.y},${neighbor.z}`)) {
            queue.push(neighbor);
          }
        }
      }
      
      if (oreBlocks.length === 0) {
        return { success: false, reason: 'no_ore' };
      }
      
      let blocksMined = 0;
      const drops: Record<string, number> = {};
      for (const pos of oreBlocks) {
        const block = mc.getBlock(pos.x, pos.y, pos.z);
        if (!block) continue;
        const blockDrops = await this.player.mineBlock(block);
        blocksMined++;
        for (const drop of blockDrops) {
          drops[drop.name] = (drops[drop.name] || 0) + drop.count;
        }
      }
      
      return {
        success: true, reason: 'success', blocksAffected: blocksMined,
        drops: Object.entries(drops).map(([name, count]) => ({ name, count }))
      };
    }
    
    return { success: false, reason: 'error' };
  }
  
  // 辅助方法
  private selectToolForBlock(block: Block): string | null {
    const blockName = block.name;
    
    // 根据方块类型选择工具
    if (blockName.includes('ore') || blockName === 'stone' || blockName === 'cobblestone') {
      return 'pickaxe';
    } else if (blockName === 'dirt' || blockName === 'grass_block' || blockName === 'sand') {
      return 'shovel';
    } else if (blockName.includes('log') || blockName.includes('planks')) {
      return 'axe';
    } else if (blockName === 'web') {
      return 'sword';
    }
    
    return null;
  }
  
  private isOre(block: Block): boolean {
    return block.name.includes('ore');
  }
}
```

| 组件 | 职责 |
|---|---|
| **LLM（大脑）** | 决定要挖/放/用方块，调用方块工具 |
| **方块工具** | 封装方块操作，返回执行结果 |
| **执行AI（假人）** | 负责具体方块操作（自动选择工具、批量操作等） |

### 5.2 关键特性

1. **自动选择工具**：执行AI自动选择合适工具（挖石头用镐、挖土用锹）
2. **批量操作**：支持区域填充、批量破坏、矿脉挖掘
3. **有反馈**：工具返回操作结果（成功/失败/详情）
4. **可失败**：工具损坏、方块不可破坏、距离太远等
