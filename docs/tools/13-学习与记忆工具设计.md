# 13-学习与记忆工具设计

## 1. 模块概述

### 1.1 目标
让 AI 能够：
- 记住玩家习惯和偏好
- 记住地图关键点（资源点、危险区域、基地）
- 记住任务经验和教训
- 记住与其他玩家的交互历史
- 通过记忆优化决策

### 1.2 架构

```
┌─────────────────────────────────────────┐
│              LLM 大脑                   │
│  - 决策时参考记忆                       │

               │
```

## 4. 工具详细设计

### 4.1 memory_store

**功能**：存储新记忆，支持单个存储和批量存储

**参数**：
```typescript
{
  memory?: {
    type: 'player_habit' | 'map_point' | 'task_experience' | 'social' | 'skill',
    content: any,              // 记忆内容（根据类型不同）
    tags?: string[],           // 标签
    importance?: number        // 重要度 (1-10，默认 5)
  },
  items?: Array<{              // batch 模式：批量存储
    type: string,
    content: any,
    tags?: string[],
    importance?: number
  }>
}
```

**返回**：
```typescript
{
  success: boolean,
  memory_id?: string,
  memory_ids?: string[],       // batch 模式返回
  error?: string
}
```

**示例**：
```typescript
// 存储单个记忆
await memory_store({
  memory: {
    type: 'player_habit',
    content: {
      player_id: 'Steve',
      habit_type: 'preference',
      description: 'Steve 喜欢用钻石工具',
      examples: ['给我钻石镐'],
      frequency: 10,
      confidence: 0.8
    },
    tags: ['Steve', '工具偏好'],
    importance: 7
  }
});

// 批量存储
await memory_store({
  items: [
    {
      type: 'map_point',
      content: {
        point_type: 'resource',
        location: { x: 128, y: 20, z: -256, dimension: 'overworld' },
        name: '钻石矿脉',
        description: '大量钻石矿',
        resources: ['diamond_ore']
      },
      tags: ['钻石', '矿区'],
      importance: 9
    },
    {
      type: 'map_point',
      content: {
        point_type: 'base',
        location: { x: 0, y: 64, z: 0, dimension: 'overworld' },
        name: '主基地'
      },
      importance: 10
    }
  ]
});
```

### 4.2 memory_recall

**功能**：按条件检索记忆，支持按 ID、类型、标签、文本查询和语义相似度检索

**参数**：
```typescript
{
  filter?: {
    id?: string,             // 按 ID 精确查找
    type?: string,           // 记忆类型
    tags?: string[],         // 标签（AND 逻辑）
    query?: string,          // 文本查询（关键词匹配）
    similar_to?: string      // 语义相似查询（向量检索）
  },
  limit?: number             // 返回数量限制
}
```

**返回**：
```typescript
{
  success: boolean,
  memories: Memory[],
  total: number
}
```

**示例**：
```typescript
// 按 ID 查询
await memory_recall({
  filter: { id: 'mem_123456' }
});

// 按类型和标签查询
await memory_recall({
  filter: {
    type: 'map_point',
    tags: ['钻石']
  },
  limit: 10
});

// 文本查询
await memory_recall({
  filter: {
    query: '钻石矿位置'
  }
});

// 语义相似查询
await memory_recall({
  filter: {
    similar_to: 'mem_123456',
    type: 'map_point'
  },
  limit: 5
});
```

## 5. 记忆存储实现

### 5.1 存储架构

```typescript
class MemoryStorage {
  private memories: Map<string, Memory> = new Map();
  private vectorIndex: VectorIndex;  // 向量索引（用于语义检索）
  private storagePath: string;
  
  constructor(config: MemoryConfig) {
    this.storagePath = config.storagePath;
    this.vectorIndex = new VectorIndex({
      dimension: 768,  // 嵌入向量维度
      metric: 'cosine'
    });
    this.loadFromDisk();
  }
  
  // 存储记忆
  async store(memory: Memory): Promise<string> {
    const memoryId = this.generateId();
    memory.id = memoryId;
    memory.created_at = Date.now();
    memory.updated_at = Date.now();
    memory.access_count = 0;
    
    this.memories.set(memoryId, memory);
    
    // 添加到向量索引
    const embedding = await this.generateEmbedding(memory);
    this.vectorIndex.add(memoryId, embedding);
    
    // 持久化
    await this.saveToDisk();
    
    return memoryId;
  }
  
  // 检索记忆
  async recall(filter: MemoryFilter): Promise<Memory[]> {
    let results = Array.from(this.memories.values());
    
    // 应用过滤条件
    if (filter.type) {
      results = results.filter(m => m.type === filter.type);
    }
    if (filter.tags && filter.tags.length > 0) {
      results = results.filter(m => 
        filter.tags!.every(tag => m.tags.includes(tag))
      );
    }
    if (filter.importance_min) {
      results = results.filter(m => m.importance >= filter.importance_min!);
    }
    if (filter.created_after) {
      results = results.filter(m => m.created_at >= filter.created_after!);
    }
    
    // 排序
    if (filter.sort_by) {
      results.sort((a, b) => {
        switch (filter.sort_by) {
          case 'importance':
            return b.importance - a.importance;
          case 'created_at':
            return b.created_at - a.created_at;
          case 'access_count':
            return b.access_count - a.access_count;
          case 'updated_at':
            return b.updated_at - a.updated_at;
          default:
            return 0;
        }
      });
    }
    
    // 限制数量
    if (filter.limit) {
      results = results.slice(0, filter.limit);
    }
    
    return results;
  }
  
  // 语义检索
  async getSimilar(query: string, options: SimilarOptions): Promise<SimilarResult[]> {
    const queryEmbedding = await this.generateEmbedding({ content: query });
    const similarIds = this.vectorIndex.search(queryEmbedding, options.limit || 5);
    
    const results: SimilarResult[] = [];
    for (const { id, similarity } of similarIds) {
      if (similarity < (options.similarity_threshold || 0.7)) continue;
      
      const memory = this.memories.get(id)!;
      if (options.type && memory.type !== options.type) continue;
      
      results.push({ memory, similarity });
    }
    
    return results.sort((a, b) => b.similarity - a.similarity);
  }
  
  // 生成嵌入向量
  private async generateEmbedding(memory: Memory | { content: any }): Promise<number[]> {
    // 将记忆内容转换为文本
    const text = this.memoryToText(memory);
    
    // 调用嵌入模型（如 OpenAI embeddings）
    const embedding = await openai.embeddings.create({
      model: 'text-embedding-3-small',
      input: text
    });
    
    return embedding.data[0].embedding;
  }
  
  // 记忆转文本
  private memoryToText(memory: Memory | { content: any }): string {
    if ('type' in memory) {
      // 完整记忆对象
      switch (memory.type) {
        case 'player_habit':
          return `${memory.content.player_id} ${memory.content.description}`;
        case 'map_point':
          return `${memory.content.name} ${memory.content.description} 位置 ${memory.content.location.x},${memory.content.location.y},${memory.content.location.z}`;
        case 'task_experience':
          return `${memory.content.description} ${memory.content.lessons.join(' ')} ${memory.content.tips.join(' ')}`;
        case 'social':
          return `${memory.content.player_id} ${memory.content.notes.join(' ')}`;
        case 'skill':
          return `${memory.content.description} ${memory.content.code || ''}`;
        default:
          return JSON.stringify(memory.content);
      }
    } else {
      // 查询文本
      return memory.content;
    }
  }
  
  // 持久化到磁盘
  private async saveToDisk(): Promise<void> {
    const data = {
      memories: Array.from(this.memories.values()),
      vectorIndex: this.vectorIndex.serialize()
    };
    
    await fs.writeFile(
      this.storagePath,
      JSON.stringify(data, null, 2)
    );
  }
  
  // 从磁盘加载
  private async loadFromDisk(): Promise<void> {
    if (!fs.existsSync(this.storagePath)) return;
    
    const data = JSON.parse(await fs.readFile(this.storagePath, 'utf-8'));
    
    // 加载记忆
    for (const memory of data.memories) {
      this.memories.set(memory.id, memory);
    }
    
    // 加载向量索引
    this.vectorIndex.deserialize(data.vectorIndex);
  }
}
```
