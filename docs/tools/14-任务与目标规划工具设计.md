# 14-任务与目标规划工具设计

## 1. 模块概述

### 1.1 目标
让 AI 能够：
- 创建和管理任务（简单/复合/循环/条件）
- 分解复杂任务为子任务
- 管理任务优先级和依赖关系
- 跟踪任务状态和进度
- 调度任务执行顺序
- 处理任务失败和重试

## 2. 任务数据结构

### 2.1 基础任务

```typescript
interface Task {
  id: string;
  name: string;                    // 任务名称
  description: string;             // 任务描述
  type: 'simple' | 'composite' | 'loop' | 'conditional';  // 任务类型
  
  // 任务状态
  status: 'pending' | 'running' | 'paused' | 'completed' | 'failed' | 'cancelled';
  progress: number;                // 进度 (0-100)
  
  // 任务内容
  action?: ToolCall;               // 简单任务：调用哪个工具
  subtasks?: string[];             // 复合任务：子任务 ID 列表
  loop_config?: LoopConfig;        // 循环任务：循环配置
  condition?: Condition;           // 条件任务：触发条件
  
  // 任务控制
  priority: 'critical' | 'high' | 'normal' | 'low';
  dependencies?: string[];         // 依赖的任务 ID
  timeout?: number;                // 超时时间（秒）
  retry_config?: RetryConfig;      // 重试配置
  
  // 任务结果
  result?: any;                    // 执行结果
  error?: string;                  // 错误信息
  retry_count?: number;            // 重试次数
  
  // 时间戳
  created_at: number;
  started_at?: number;
  completed_at?: number;
  updated_at: number;
  
  // 元数据
  tags?: string[];
  metadata?: Record<string, any>;
}

// 工具调用
interface ToolCall {
  tool_name: string;
  parameters: Record<string, any>;
}

// 循环配置
interface LoopConfig {
  mode: 'count' | 'interval' | 'condition';  // 循环模式
  count?: number;                  // 循环次数（mode='count'）
  interval?: number;               // 间隔时间（秒，mode='interval'）
  condition?: string;              // 循环条件（mode='condition'）
  max_iterations?: number;         // 最大迭代次数（防止无限循环）
}

// 条件
interface Condition {
  type: 'time' | 'event' | 'state' | 'expression';
  value: any;                      // 条件值
}

// 重试配置
interface RetryConfig {
  max_retries: number;             // 最大重试次数
  retry_delay: number;             // 重试间隔（秒）
  backoff_multiplier?: number;     // 退避乘数（指数退避）
}
```

## 5. 任务调度器实现

### 5.1 优先级队列

```typescript
class TaskScheduler {
  private taskQueue: PriorityQueue<Task>;
  private runningTasks: Map<string, Task> = new Map();
  private maxConcurrent: number = 3;  // 最大并发任务数
  
  constructor() {
    this.taskQueue = new PriorityQueue<Task>((a, b) => {
      // 优先级比较
      const priorityOrder = { critical: 0, high: 1, normal: 2, low: 3 };
      const priorityDiff = priorityOrder[a.priority] - priorityOrder[b.priority];
      if (priorityDiff !== 0) return priorityDiff;
      
      // 相同优先级按创建时间排序
      return a.created_at - b.created_at;
    });
  }
  
  // 添加任务到队列
  async enqueue(task: Task): Promise<void> {
    // 检查依赖
    if (task.dependencies && task.dependencies.length > 0) {
      const unmetDeps = await this.checkDependencies(task.dependencies);
      if (unmetDeps.length > 0) {
        task.status = 'pending';
        task.metadata = { ...task.metadata, waiting_for: unmetDeps };
      }
    }
    
    this.taskQueue.enqueue(task);
  }
  
  // 调度循环
  async schedule(): Promise<void> {
    while (true) {
      // 检查是否有可以执行的任务
      if (this.runningTasks.size < this.maxConcurrent && !this.taskQueue.isEmpty()) {
        const task = this.taskQueue.peek();
        
        // 检查依赖是否满足
        if (task.dependencies && task.dependencies.length > 0) {
          const unmetDeps = await this.checkDependencies(task.dependencies);
          if (unmetDeps.length > 0) {
            // 依赖未满足，跳过
            this.taskQueue.dequeue();
            this.taskQueue.enqueue(task);  // 放回队列末尾
            continue;
          }
        }
        
        // 执行任务
        this.taskQueue.dequeue();
        this.executeTask(task);
      }
      
      // 等待一段时间再检查
      await sleep(1000);
    }
  }
  
  // 执行任务
  private async executeTask(task: Task): Promise<void> {
    task.status = 'running';
    task.started_at = Date.now();
    this.runningTasks.set(task.id, task);
    
    try {
      let result: any;
      
      switch (task.type) {
        case 'simple':
          result = await this.executeSimpleTask(task);
          break;
        case 'composite':
          result = await this.executeCompositeTask(task);
          break;
        case 'loop':
          result = await this.executeLoopTask(task);
          break;
        case 'conditional':
          result = await this.executeConditionalTask(task);
          break;
      }
      
      // 任务成功
      task.status = 'completed';
      task.progress = 100;
      task.result = result;
      task.completed_at = Date.now();
      
      // 触发后续任务
      await this.triggerDependentTasks(task.id);
      
    } catch (error) {
      // 任务失败
      task.error = error.message;
      
      // 检查是否可以重试
      if (task.retry_config && (task.retry_count || 0) < task.retry_config.max_retries) {
        task.retry_count = (task.retry_count || 0) + 1;
        task.status = 'pending';
        
        // 延迟重试
        const delay = task.retry_config.retry_delay * 
                     (task.retry_config.backoff_multiplier || 1) ** (task.retry_count - 1);
        setTimeout(() => this.enqueue(task), delay * 1000);
      } else {
        task.status = 'failed';
        task.completed_at = Date.now();
      }
    } finally {
      this.runningTasks.delete(task.id);
    }
  }
  
  // 执行简单任务
  private async executeSimpleTask(task: Task): Promise<any> {
    if (!task.action) throw new Error('Simple task must have an action');
    
    const { tool_name, parameters } = task.action;
    const tool = getTool(tool_name);
    
    return await tool.execute(parameters);
  }
  
  // 执行复合任务
  private async executeCompositeTask(task: Task): Promise<any> {
    if (!task.subtasks || task.subtasks.length === 0) {
      throw new Error('Composite task must have subtasks');
    }
    
    const results = [];
    for (const subtaskId of task.subtasks) {
      const subtask = await getTaskById(subtaskId);
      const result = await this.executeTask(subtask);
      results.push(result);
      
      // 更新进度
      task.progress = (results.length / task.subtasks.length) * 100;
    }
    
    return results;
  }
  
  // 执行循环任务
  private async executeLoopTask(task: Task): Promise<any> {
    if (!task.loop_config) throw new Error('Loop task must have loop_config');
    
    const { mode, count, interval, condition, max_iterations } = task.loop_config;
    const results = [];
    let iteration = 0;
    
    while (true) {
      // 检查最大迭代次数
      if (max_iterations && iteration >= max_iterations) break;
      
      // 根据模式判断是否继续
      if (mode === 'count' && iteration >= count!) break;
      if (mode === 'condition' && !evaluateCondition(condition!)) break;
      
      // 执行任务
      const result = await this.executeSimpleTask(task);
      results.push(result);
      iteration++;
      
      // 更新进度
      if (mode === 'count') {
        task.progress = (iteration / count!) * 100;
      }
      
      // 等待间隔
      if (mode === 'interval' && interval) {
        await sleep(interval * 1000);
      }
    }
    
    return results;
  }
  
  // 执行条件任务
  private async executeConditionalTask(task: Task): Promise<any> {
    if (!task.condition) throw new Error('Conditional task must have a condition');
    
    // 等待条件满足
    while (!evaluateCondition(task.condition)) {
      await sleep(1000);
    }
    
    // 条件满足，执行任务
    return await this.executeSimpleTask(task);
  }
  
  // 检查依赖
  private async checkDependencies(dependencies: string[]): Promise<string[]> {
    const unmet = [];
    for (const depId of dependencies) {
      const depTask = await getTaskById(depId);
      if (depTask.status !== 'completed') {
        unmet.push(depId);
      }
    }
    return unmet;
  }
  
  // 触发后续任务
  private async triggerDependentTasks(taskId: string): Promise<void> {
    const allTasks = await getAllTasks();
    for (const task of allTasks) {
      if (task.dependencies && task.dependencies.includes(taskId)) {
        // 检查所有依赖是否都完成
        const unmetDeps = await this.checkDependencies(task.dependencies);
        if (unmetDeps.length === 0) {
          // 所有依赖都完成，加入队列
          await this.enqueue(task);
        }
      }
    }
  }
}
```

## 6. 使用场景

### 6.1 场景 1：复杂任务分解

```
用户请求: "建造一个村民交易所"
  ↓
LLM 调用: task_decompose({ task_description: "建造村民交易所" })
  ↓
返回子任务:
  1. 收集建筑材料
  2. 建造房屋框架
  3. 捕获村民
  4. 设置交易站
  ↓
LLM 创建复合任务:
  task_create({
    name: "建造村民交易所",
    type: "composite",
    subtasks: ["task_001", "task_002", "task_003", "task_004"]
  })
  ↓
调度器自动执行子任务（按依赖关系）
  ↓
任务完成，通知用户
```

### 6.2 场景 2：循环巡逻

```
LLM 创建循环任务:
  task_create({
    name: "巡逻基地",
    type: "loop",
    action: { tool_name: "move_to", parameters: { path: "patrol_route" } },
    loop_config: { mode: "interval", interval: 300 }
  })
  ↓
调度器每 5 分钟执行一次巡逻
  ↓
巡逻过程中发现怪物
  ↓
触发条件任务: "击杀怪物"
  ↓
继续巡逻
```

### 6.3 场景 3：条件触发

```
LLM 创建条件任务:
  task_create({
    name: "饥饿时吃东西",
    type: "conditional",
    action: { tool_name: "eat", parameters: {} },
    condition: { type: "state", value: "hunger < 10" }
  })
  ↓
调度器持续监控饥饿值
  ↓
饥饿值降到 10 以下
  ↓
自动执行吃东西任务
  ↓
饥饿值恢复，任务暂停等待下次触发
```

### 6.4 场景 4：任务依赖

```
任务 A: 收集木材 (无依赖)
任务 B: 建造墙壁 (依赖 A)
任务 C: 建造屋顶 (依赖 B)
  ↓
调度器先执行 A
  ↓
A 完成，触发 B
  ↓
B 完成，触发 C
  ↓
C 完成，复合任务完成
```
